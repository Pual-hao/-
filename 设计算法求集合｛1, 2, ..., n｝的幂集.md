利用递归算法
设计算法求集合｛1, 2, ..., n｝的幂集。
![image](https://user-images.githubusercontent.com/94223306/166094483-9b6a7fe4-b5b2-4558-ad1c-4edafa2059fd.png)

```cpp
#include <iostream>
#include<vector>
std::vector<std::vector<int>>b;
void func(int a)
{
    if(a==0)
    {
        std::vector<int>c;
        b.push_back(c);
        return;
    }
    func(a-1);
    int size=b.size();
    for(int i=0;i<size;i++)
    {
        std::vector<int>c(b.at(i));
        c.push_back(a);
        b.push_back(c);
    }
}
int main() {
    int n;
    std::cin>>n;
    func(n);
    for(int i=0;i<b.size();i++)
    {
        std::cout<<'{';
        for(int j=0;j<b[i].size();j++)
        {
            std::cout<<b[i][j]<<' ';
        }
        std::cout<<'}'<<std::endl;
    }
    return 0;
}
```
